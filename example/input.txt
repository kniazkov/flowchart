BEGIN
INPUT read m
INPUT read n
IF n > 0 or m > 0  ' check correctness
    :label
    IF m = n
        OUTPUT write m
        END
    ELSE
        IF m > n
            m = m - n
        ELSE
            n = n - m
        END_IF
        GOTO label
    END_IF
ELSE
    ERROR incorrect values
END_IF
